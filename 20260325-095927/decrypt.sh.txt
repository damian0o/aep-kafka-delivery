#!/usr/bin/env bash
#
# Decrypt a delivery bundle file.
#
# Usage:
#   ./decrypt.sh <passphrase> <bundle-file> [output-dir]
#
# The bundle file contains encrypted files concatenated as:
#   ===FILE:<filename>===
#   <base64 encrypted content>
#   ===END===
#
# Each section is base64-decoded, then decrypted with AES-256-CBC.
#
set -euo pipefail

PASSPHRASE="${1:?Usage: $0 <passphrase> <bundle-file> [output-dir]}"
BUNDLE_FILE="${2:?Usage: $0 <passphrase> <bundle-file> [output-dir]}"
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
OUTPUT_DIR="${3:-$SCRIPT_DIR/decrypted}"

if [ ! -f "$BUNDLE_FILE" ]; then
    echo "Error: bundle file not found: $BUNDLE_FILE"
    exit 1
fi

mkdir -p "$OUTPUT_DIR"

CURRENT_FILE=""
CONTENT=""

while IFS= read -r line; do
    if [[ "$line" == ===FILE:*=== ]]; then
        CURRENT_FILE="${line#===FILE:}"
        CURRENT_FILE="${CURRENT_FILE%===}"
        CONTENT=""
    elif [[ "$line" == "===END===" ]]; then
        if [ -n "$CURRENT_FILE" ]; then
            echo "$CONTENT" | base64 -d \
                | openssl enc -aes-256-cbc -d -salt -pbkdf2 -iter 100000 \
                    -out "$OUTPUT_DIR/$CURRENT_FILE" \
                    -pass "pass:$PASSPHRASE"
            [[ "$CURRENT_FILE" == *.sh ]] && chmod +x "$OUTPUT_DIR/$CURRENT_FILE"
            echo "Decrypted: $CURRENT_FILE"
        fi
        CURRENT_FILE=""
        CONTENT=""
    elif [ -n "$CURRENT_FILE" ]; then
        if [ -z "$CONTENT" ]; then
            CONTENT="$line"
        else
            CONTENT="$CONTENT
$line"
        fi
    fi
done < "$BUNDLE_FILE"

echo ""
echo "Files decrypted to: $OUTPUT_DIR"
echo "To install: cd <target-repo> && $OUTPUT_DIR/install.sh"
